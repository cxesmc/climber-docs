program climber

  use atm_model, only: atm_init, atm_update, atm_end, atm_write_restart
  use atm_def, only: atm_class
  use atm_out, only: atm_diag, atm_diag_init

  use coupler, only: cmn_class
  use coupler, only: atm_to_cmn, cmn_to_atm

  ! define derived types
  type(cmn_class) :: cmn    ! common grid (coupler)
  type(atm_class) :: atm    ! atmosphere

  ! Initialise atmosphere model
  call atm_init(...)
  call atm_diag_init

  ! time loop
  do step=1,nstep

    call timer_update

    ! update atmosphere model
    call cmn_to_atm(cmn,atm)      ! transfer data from coupler to the atmosphere
    call atm_update(atm)          ! update atmosphere model
    call atm_diag(atm)            ! atmosphere diagnostics
    call atm_to_cmn(atm,cmn)      ! transfer data from atmosphere to coupler

  enddo

  ! write restart file
  call atm_write_restart(trim(rest_dir)//"/atm_restart.nc",atm)

  ! End atmosphere model
  call atm_end(atm)

end program climber
